# Phase 3 — MinIO Object Storage

## Overview

Replace Supabase Storage with MinIO — a self-hosted, S3-compatible object storage system. MinIO handles PDF uploads, avatar images, and file exports. All document access goes through presigned URLs generated by the Python backend.

**Duration**: Week 2 (parallel with auth)  
**Dependencies**: Phase 1 (MinIO running in Docker)  
**Deliverables**: Storage service, upload/download API, presigned URLs, frontend upload refactored

---

## Step-by-Step Guide

### Step 3.1 — Create MinIO Client Module

```python
# app/core/minio_client.py
from minio import Minio
from app.config import settings

minio_client = Minio(
    settings.MINIO_ENDPOINT,
    access_key=settings.MINIO_ACCESS_KEY,
    secret_key=settings.MINIO_SECRET_KEY,
    secure=settings.MINIO_SECURE,
)

def init_buckets():
    """Create required buckets on startup."""
    buckets = [
        settings.MINIO_BUCKET_DOCUMENTS,
        settings.MINIO_BUCKET_AVATARS,
    ]
    for bucket in buckets:
        if not minio_client.bucket_exists(bucket):
            minio_client.make_bucket(bucket)
```

### Step 3.2 — Create Storage Service

```python
# app/services/storage_service.py
import io
import uuid
from datetime import datetime, timedelta
from minio import Minio
from app.core.minio_client import minio_client
from app.config import settings

class StorageService:
    def __init__(self):
        self.client = minio_client

    async def upload_document(
        self, user_id: str, file_data: bytes, filename: str, content_type: str
    ) -> str:
        """Upload a PDF document to MinIO. Returns the object path."""
        timestamp = int(datetime.now().timestamp())
        object_name = f"{user_id}/{timestamp}_{filename}"

        self.client.put_object(
            bucket_name=settings.MINIO_BUCKET_DOCUMENTS,
            object_name=object_name,
            data=io.BytesIO(file_data),
            length=len(file_data),
            content_type=content_type,
        )
        return object_name

    async def get_document_url(self, object_name: str, expires: int = 3600) -> str:
        """Generate a presigned download URL (default: 1 hour)."""
        return self.client.presigned_get_object(
            bucket_name=settings.MINIO_BUCKET_DOCUMENTS,
            object_name=object_name,
            expires=timedelta(seconds=expires),
        )

    async def delete_document(self, object_name: str):
        """Delete a document from MinIO."""
        self.client.remove_object(
            bucket_name=settings.MINIO_BUCKET_DOCUMENTS,
            object_name=object_name,
        )

    async def get_document_bytes(self, object_name: str) -> bytes:
        """Download document bytes for server-side processing."""
        response = self.client.get_object(
            bucket_name=settings.MINIO_BUCKET_DOCUMENTS,
            object_name=object_name,
        )
        try:
            return response.read()
        finally:
            response.close()
            response.release_conn()

    async def upload_avatar(
        self, user_id: str, file_data: bytes, content_type: str
    ) -> str:
        """Upload an avatar image. Returns the public URL path."""
        object_name = f"{user_id}/avatar"

        self.client.put_object(
            bucket_name=settings.MINIO_BUCKET_AVATARS,
            object_name=object_name,
            data=io.BytesIO(file_data),
            length=len(file_data),
            content_type=content_type,
        )
        return object_name
```

### Step 3.3 — Create Document Upload API Endpoint

```python
# app/api/documents.py (upload section)
from fastapi import APIRouter, Depends, UploadFile, File, HTTPException
from sqlalchemy.ext.asyncio import AsyncSession
from app.core.database import get_db
from app.core.security import get_current_user
from app.services.storage_service import StorageService

router = APIRouter(prefix="/api/documents", tags=["documents"])
storage_service = StorageService()

ALLOWED_MIME_TYPES = {"application/pdf"}
MAX_FILE_SIZE = 50 * 1024 * 1024  # 50MB

@router.post("/upload", status_code=201)
async def upload_document(
    file: UploadFile = File(...),
    user: dict = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
    """Upload a PDF document."""
    # Validate MIME type
    if file.content_type not in ALLOWED_MIME_TYPES:
        raise HTTPException(status_code=400, detail="Only PDF files are allowed")

    # Read file
    file_data = await file.read()

    # Validate file size
    if len(file_data) > MAX_FILE_SIZE:
        raise HTTPException(status_code=400, detail="File too large (max 50MB)")

    # Upload to MinIO
    file_path = await storage_service.upload_document(
        user_id=user["id"],
        file_data=file_data,
        filename=file.filename or "document.pdf",
        content_type=file.content_type,
    )

    # Create database record
    doc_id = str(uuid.uuid4())
    await db.execute(
        text("""
            INSERT INTO documents (id, user_id, title, file_path, file_size, status)
            VALUES (:id, :user_id, :title, :file_path, :file_size, 'processing')
        """),
        {
            "id": doc_id,
            "user_id": user["id"],
            "title": file.filename or "Untitled",
            "file_path": file_path,
            "file_size": len(file_data),
        },
    )

    # TODO: Trigger Celery task for text extraction + embedding (Phase 5)

    return {"id": doc_id, "title": file.filename, "status": "processing"}

@router.get("/{document_id}/download-url")
async def get_download_url(
    document_id: str,
    user: dict = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
    """Get a presigned download URL for a document."""
    # Verify ownership
    result = await db.execute(
        text("SELECT file_path FROM documents WHERE id = :id AND user_id = :user_id"),
        {"id": document_id, "user_id": user["id"]},
    )
    row = result.fetchone()
    if not row:
        raise HTTPException(status_code=404, detail="Document not found")

    url = await storage_service.get_document_url(row.file_path)
    return {"url": url}
```

### Step 3.4 — Set Bucket Policies

The `documents` bucket stays **private** — all access through presigned URLs.
The `avatars` bucket is **public read**:

```python
# Add to init_buckets() in minio_client.py
import json

def init_buckets():
    buckets = [settings.MINIO_BUCKET_DOCUMENTS, settings.MINIO_BUCKET_AVATARS]
    for bucket in buckets:
        if not minio_client.bucket_exists(bucket):
            minio_client.make_bucket(bucket)

    # Set avatars bucket to public read
    policy = {
        "Version": "2012-10-17",
        "Statement": [
            {
                "Effect": "Allow",
                "Principal": {"AWS": ["*"]},
                "Action": ["s3:GetObject"],
                "Resource": [f"arn:aws:s3:::{settings.MINIO_BUCKET_AVATARS}/*"],
            }
        ],
    }
    minio_client.set_bucket_policy(
        settings.MINIO_BUCKET_AVATARS, json.dumps(policy)
    )
```

## MinIO API Mapping

| Supabase Storage | MinIO Equivalent |
|-----------------|------------------|
| `supabase.storage.from('documents').upload(path, file)` | `minio_client.put_object(bucket, path, data, length)` |
| `supabase.storage.from('documents').getPublicUrl(path)` | `minio_client.presigned_get_object(bucket, path, expires)` |
| `supabase.storage.from('documents').download(path)` | `minio_client.get_object(bucket, path)` |
| `supabase.storage.from('documents').remove([path])` | `minio_client.remove_object(bucket, path)` |
| Storage RLS policies | Backend-level ownership checks |
| 50MB file size limit | Configurable (Nginx/MinIO) |

## Common Issues

1. **MinIO connection refused**: Ensure MinIO is running and `MINIO_ENDPOINT` matches Docker service name (`minio:9000` inside Docker, `localhost:9000` outside)
2. **Presigned URL not working**: Check `MINIO_SECURE` setting — use `false` for local dev (HTTP), `true` for production (HTTPS)
3. **CORS on presigned URLs**: If the browser blocks presigned URL downloads, configure MinIO CORS or proxy through Nginx
4. **Large files slow**: For production, consider Nginx upload buffering or MinIO multipart uploads
